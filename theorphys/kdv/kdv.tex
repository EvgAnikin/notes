	\noindent Уравнение КдФ имеет вид
	\begin{equation}
		u_t + 6uu_x + u_{xxx} = 0
	\end{equation}
	\begin{theorem}[Галилеевская инвариантность]
		Пусть $u(t,x)$ --- решение КдФ. Тогда $u(t, x - vt) + \frac{v}{6}$ --- 
		тоже решение КдФ.
	\end{theorem}
	Проверяется элементарной подстановкой.
